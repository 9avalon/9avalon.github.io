美团金融的一道面试题

很明显，超大数用Integer或者Long来装是不行的，必须要用String来进行存储，这里我采用了最笨的方法，就是模拟乘法的原理，去进行计算。

```java
    public static void twoBigNumberAdd(String leftNum, String rightNum) {
        // 转化成数组，并逆向排序
        Integer[] leftCharArray = new Integer[leftNum.length()];
        Integer[] rightCharArray = new Integer[rightNum.length()];
        for (int i = 0; i < leftCharArray.length; i++) {
            leftCharArray[i] = leftNum.charAt(leftNum.length() - i - 1) - '0';
        }
        for (int i = 0; i < rightCharArray.length; i++) {
            rightCharArray[i] = rightNum.charAt(rightNum.length() - i - 1) - '0';
        }
        
        // 逐位相乘进行计算
        int[] allCharArray = new int[leftNum.length() + rightNum.length()];
        for (int i = 0; i < leftCharArray.length; i++) {
            for (int j = 0; j < rightCharArray.length; j++) {
                allCharArray[i + j] += leftCharArray[i] * rightCharArray[j];
            }
        }
        
        // 将计算后的结果进行逐步移位
        for (int i = 0; i < allCharArray.length - 1; i++) {
            Integer addNum = allCharArray[i];
            allCharArray[i + 1] += addNum / 10;
            allCharArray[i] = addNum - (addNum / 10) * 10;
        }
        
        // 转换成String，并且去掉多余的0
        StringBuffer sb = new StringBuffer();
        boolean zero = false;
        for (int i = 0; i < allCharArray.length; i++) {
            if (allCharArray[allCharArray.length - 1 - i] != 0) {
                zero = true;
            }
            if (zero == true){
                sb.append(allCharArray[allCharArray.length - 1 - i]);
            }
        }

        System.out.println(sb);
    }
```
